Punto 1:

"Realizar la siguiente figura por medio del ciclo Para"

                
                    Proceso cicloParaFigura
                        Definir inicio Como Entero;
                        Definir asterisco Como Entero;
                        
                        Para inicio <-1 Hasta 10 Hacer
                            Para asterisco <- 1 Hasta inicio Hacer

                            FinPara

                        FinPara
                    FinProceso

Punto 2:

"Realizar la siguiente figura por medio del ciclo Mientras - Hacer"

                
                    Proceso cicloMientrasFiguraInvertida
                        Definir global Como Entero;
                        Definir espacio Como Entero;
                        Definir asterisco Como Entero;
                        
                        global <- 10;
                        
                        Mientras global >= 1 Hacer
                            espacio <- 1;
                            Mientras espacio <= global hacer			
                                Escribir Sin Saltar " " ;
                                espacio <- espacio+1;
                            FinMientras
                            
                            asterisco <- 10;
                            Mientras asterisco >= global Hacer
                                Escribir  "*" Sin Saltar;
                                asterisco <- asterisco-1;
                            FinMientras
                            
                            Escribir "";
                            global <- global - 1;		
                        FinMientras	
                    FinProceso

                                        en el parqueadero';
                                    FinSi
                                3:	
                                    Escribir Sin Saltar 'Ingresa la placa del vehículo a retirar: ';

                                        placa0 <- 'Retirado';
